```python
    with xw.App(visible=False, add_book=False) as app:
        wb = app.books.open(file_path)
        app.calculation = 'automatic'
        wb.app.calculate()
        for sheet in wb.sheets:
            if sheet.name in SHEETS_CONFIG:
                sheet.calculate()
        wb.save()
        wb.close()
```
